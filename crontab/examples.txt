#Exemplos do configuracoes para o crontab
#Some examples of crontab configurations
0 0 * * * echo IT WORKS! > /home/test.txt # will, once per day at 00:00, create a .txt file named test if it doe not exists, and put "IT WORKS!" on the first line.
0 0 * * * echo FUNCIONA! > /home/teste.txt # uma vez ao dia, as 00:00, cria um arquivo .txt de nome teste, caso este nao exista, e grava na primeira linha "FUNCIONA!".
